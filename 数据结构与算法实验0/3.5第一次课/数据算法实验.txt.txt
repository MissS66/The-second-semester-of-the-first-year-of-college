1.车牌号问题：四位数，前俩数字一样，后俩数字一样，不为同一数字，且是某整数平方：
方法：
1.对数据进行循环自增
2.对数据进行四次次取余
3.对数据取整以便后续取余
4.对取余结果进行比对，用判断语句判断是否为真
5.下一轮循环判断上一轮循环的结果是否为一个整数（两位数）的平方
6.是的话输出
自己写的算法程序：
#include<iostream>
#include<cstdlib>
using namespace std;
int main() {
int nx=999;
int na=0, nb=0,ns=0;
for (nx; nx <= 9999; nx++) {
    int nc=0, nd=0,ne=0,nf=0;
    nc = nx % 10;//最后一位
    nd = nx / 10;//为求倒数第二位数做准备
    ne = nd % 10;//倒数第二位
    nf = nd / 10;//为求倒数第三位数做准备
    if (ne == nc) { int nm = 0, nn = 0, no = 0;
    nm = nf % 10;//倒数第三位
    nn = nf / 10;
    no = nn % 10;//倒数第四位
    if (nm == no && nm != ne)
    {   
        /*ns = nx;
        cout << ns << endl;*///输出所有的满足前两个条件的数，中途查错
        for (int ni = 31; ni < 100; ni++) {
            int nz = 0;
            nz = ni * ni;//循环平方，找相等
            if (nz == nx) {
                cout << "车牌号是："<<nx <<","<<ni << endl;
            }
        }
    }
    }
}
return 0;
}
结果：7744，是88的平方
